2. 成績データの統計計算プログラム (C言語)

■ 概要

複数の生徒における「英語」「数学」「物理」の3教科のテスト点数を2次元配列を用いて効率的に管理し、平均点、分散、最高点、最低点を算出するプログラムです。全体の処理をmain関数に直書きするのではなく、それぞれの計算を行う専用の関数を定義して処理を分割（モジュール化）しています。マクロ定義を活用して生徒数の上限変更に容易に対応できるようにし、さらに計算処理の関数の引数にはconstを付与することでデータの意図しない書き換えを防止するなど、実践的な開発に通じる保守性や安全性を強く意識したコード設計になっています。

■ 使用技術・工夫点

  • C言語
  • 2次元配列によるデータ構造の管理
  • 機能ごとの関数化（モジュール分割）
  • マクロ定義 (#define) を用いた保守性の向上
  • const 修飾子を用いた意図しないデータ書き換えの防止

#include <stdio.h>

#define MAX_STUDENTS 100
#define SUB_ENGLISH 0
#define SUB_MATH    1
#define SUB_PHYSICS 2

double calc_average(const int scores[][3], int n, int subject)
{
    int i;
    int sum = 0;
    for (i = 0; i < n; i++) {
        sum += scores[i][subject];
    }
    return (double)sum / n;
}

double calc_variance(const int scores[][3], int n, int subject, double average)
{
    int i;
    double sum_squared_diff = 0.0;
    double diff;
    
    for (i = 0; i < n; i++) {
        diff = scores[i][subject] - average;
        sum_squared_diff += diff * diff;
    }
    return sum_squared_diff / n;
}

int calc_max(const int scores[][3], int n, int subject)
{
    int i;
    int max_val = scores[0][subject]; 
    
    for (i = 1; i < n; i++) {
        if (scores[i][subject] > max_val) {
            max_val = scores[i][subject];
        }
    }
    return max_val;
}

int calc_min(const int scores[][3], int n, int subject)
{
    int i;
    int min_val = scores[0][subject]; 
    
    for (i = 1; i < n; i++) {
        if (scores[i][subject] < min_val) {
            min_val = scores[i][subject];
        }
    }
    return min_val;
}

int main(void)
{
    int n, i;
    int scores[MAX_STUDENTS][3]; 
    char *subject_names[] = {"英語", "数学", "物理"};
    int subject;

    printf("生徒数を入力してください: ");
    scanf("%d", &n);

    if (n > MAX_STUDENTS) {
        printf("生徒数が多すぎます。\n");
        return 1;
    }

    for (i = 0; i < n; i++) {
        printf("%d人目の点数 (英語 数学 物理): ", i + 1);
        scanf("%d %d %d", &scores[i][0], &scores[i][1], &scores[i][2]);
    }

    for (subject = 0; subject < 3; subject++) {
        double avg = calc_average(scores, n, subject);
        double var = calc_variance(scores, n, subject, avg);
        int max = calc_max(scores, n, subject);
        int min = calc_min(scores, n, subject);

        printf("\n--- 教科: %s ---\n", subject_names[subject]);
        printf("平均点: %.2f\n", avg);
        printf("分散: %.2f\n", var);
        printf("最高点: %d\n", max);
        printf("最低点: %d\n", min);
    }

    return 0;
}
