1. キーボード紹介デスクトップアプリ (Java)

■ 概要

Javaの標準ライブラリである「Swing」を用いて開発した、ユーザーに最適なキーボードを提案するデスクトップアプリケーションです。最初の画面で「ゲームで使う」「コンパクト」「オシャレ」といった使用目的をチェックボックス形式で複数選択させ、次の画面で「高い」などの4段階の価格帯をボタンで指定させます。入力された条件の組み合わせをプログラム内部で判定し、条件に合致する具体的なキーボードの製品名とその画像を画面中央に動的にリサイズして表示します。また、画面遷移に合わせて自動で音声ファイルが再生される機能も実装し、UIの使いやすさや視覚・聴覚的な演出にもこだわって設計しました。

■ 使用技術・工夫点

  • Java (Swing, AWT) を用いたGUI開発
  • BorderLayoutとBoxLayoutを組み合わせた画面レイアウト
  • 選択条件の組み合わせに応じた動的な画面遷移ロジック
  • AudioSystemを利用したBGM・効果音の再生機能
  • 画像サイズの自動スケーリング処理

import java.awt.*;
import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import javax.sound.sampled.*;
import java.io.File;
import java.io.IOException;

public class BasicSystem extends JFrame {
    JPanel TitlePanel = null;
    JPanel FacePanel = null;
    JPanel SelectPanel = null;

    public static void main(String[] args) {
        new BasicSystem();
    }

    public BasicSystem() {
        this.setTitle("キーボード紹介");
        this.setSize(900, 700);
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        this.setVisible(true);
        showMainPage();
    }

    public void showMainPage() {
        // 既存パネルの削除
        if (TitlePanel != null) {
            this.remove(TitlePanel);
            this.remove(FacePanel);
            this.remove(SelectPanel);
        }

        // 新しいパネルの作成
        TitlePanel = new JPanel();
        FacePanel = new JPanel(new BorderLayout()); // 中央揃えにするためのレイアウトに変更
        SelectPanel = new JPanel();
        SelectPanel.setLayout(new BoxLayout(SelectPanel, BoxLayout.Y_AXIS));

        // タイトルラベル
        JLabel label = new JLabel("<html><font size=7>キーボード紹介をするよ</font></html>");
        TitlePanel.add(label);

        // 画像の表示
        ImageIcon icon = new ImageIcon("標準.png");
        Image image0 = icon.getImage();
        Image newimg0 = image0.getScaledInstance(500, 300, java.awt.Image.SCALE_SMOOTH);
        icon = new ImageIcon(newimg0);
        JLabel photo = new JLabel(icon);
        FacePanel.add(photo, BorderLayout.CENTER); // 画像を中央に配置

        // 「あなたの目的を選んでね」のメッセージラベルを画像の下に追加
        JLabel instructionLabel = new JLabel("<html><font size=5>あなたの目的を選んでね</font></html>");
        instructionLabel.setHorizontalAlignment(SwingConstants.CENTER);
        FacePanel.add(instructionLabel, BorderLayout.SOUTH);

        // チェックボックスの作成
        JCheckBox checkBox1 = new JCheckBox("ゲームで使う");
        JCheckBox checkBox2 = new JCheckBox("コンパクト");
        JCheckBox checkBox3 = new JCheckBox("オシャレ");

        // 中央揃えのパネルにチェックボックスを追加
        JPanel checkBoxPanel = new JPanel();
        checkBoxPanel.setLayout(new BoxLayout(checkBoxPanel, BoxLayout.Y_AXIS));
        checkBoxPanel.add(checkBox1);
        checkBoxPanel.add(checkBox2);
        checkBoxPanel.add(checkBox3);
        checkBoxPanel.setAlignmentX(Component.CENTER_ALIGNMENT);

        SelectPanel.add(checkBoxPanel);

        // 次のページへ進むボタンとアクションリスナー
        JButton button = new JButton("Next");
        button.setAlignmentX(0.5f);
        button.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                boolean isSelected1 = checkBox1.isSelected();
                boolean isSelected2 = checkBox2.isSelected();
                boolean isSelected3 = checkBox3.isSelected();

                // 選択に基づいて新しいページを開く
                if (isSelected1 && !isSelected2 && !isSelected3) {
                    openNewPage("ゲームで使うが選ばれました", "Option1");
                } else if (!isSelected1 && isSelected2 && !isSelected3) {
                    openNewPage("コンパクトが選ばれました", "Option2");
                } else if (!isSelected1 && !isSelected2 && isSelected3) {
                    openNewPage("オシャレが選ばれました", "Option3");
                } else if (isSelected1 && isSelected2 && !isSelected3) {
                    openNewPage("ゲームで使うとコンパクトが選ばれました", "Option1_2");
                } else if (isSelected1 && !isSelected2 && isSelected3) {
                    openNewPage("ゲームで使うとオシャレが選ばれました", "Option1_3");
                } else if (!isSelected1 && isSelected2 && isSelected3) {
                    openNewPage("コンパクトとオシャレが選ばれました", "Option2_3");
                } else if (isSelected1 && isSelected2 && isSelected3) {
                    openNewPage("ゲームで使う、コンパクト、オシャレが選ばれました", "Option1_2_3");
                } else {
                    JOptionPane.showMessageDialog(null, "Please select only one or valid combinations.");
                }
            }
        });

        // ボタンを中央揃えのパネルに追加
        JPanel buttonPanel = new JPanel();
        buttonPanel.setAlignmentX(0.5f);
        buttonPanel.add(button);
        SelectPanel.add(buttonPanel);

        // レイアウト設定
        this.setLayout(new BorderLayout(5, 5));
        this.add(TitlePanel, BorderLayout.NORTH);
        this.add(FacePanel, BorderLayout.CENTER);
        this.add(SelectPanel, BorderLayout.SOUTH);

        this.repaint();
        this.validate();

        // メインページで自動的に音声を再生
        playSound("8.wav");
    }

    private void openNewPage(String pageTitle, String optionType) {

        newFrame.setSize(900, 700);
        newFrame.setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
        newFrame.setLocationRelativeTo(null);

        // 新しいパネルを作成
        JPanel newTitlePanel = new JPanel();
        JPanel newFacePanel = new JPanel(new BorderLayout()); // 画像とメッセージの中央揃え
        JPanel newSelectPanel = new JPanel();
        newSelectPanel.setLayout(new BoxLayout(newSelectPanel, BoxLayout.Y_AXIS));

        // タイトルラベル
        JLabel newLabel = new JLabel("<html><font size=7>価格帯を選んでください</font></html>");
        newTitlePanel.add(newLabel);

        // 画像の表示
        ImageIcon newIcon = new ImageIcon("標準.png");
        Image newImage0 = newIcon.getImage();
        Image newResizedImg0 = newImage0.getScaledInstance(500, 300, java.awt.Image.SCALE_SMOOTH);
        newIcon = new ImageIcon(newResizedImg0);
        JLabel newPhoto = new JLabel(newIcon);
        newFacePanel.add(newPhoto, BorderLayout.CENTER);

        // 選択目的の表示
        JLabel selectedPurposeLabel = new JLabel("<html><font size=5>あなたが選んだ目的: " + pageTitle + "</font></html>");
        selectedPurposeLabel.setHorizontalAlignment(SwingConstants.CENTER);
        newFacePanel.add(selectedPurposeLabel, BorderLayout.SOUTH);

        // 価格帯選択ボタンの作成
        JButton optionAButton = new JButton("高い");
        optionAButton.setAlignmentX(Component.CENTER_ALIGNMENT);
        optionAButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                showResult(optionType, "A");
            }
        });
        newSelectPanel.add(optionAButton);

        JButton optionBButton = new JButton("ちょい高い");
        optionBButton.setAlignmentX(Component.CENTER_ALIGNMENT);
        optionBButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                showResult(optionType, "B");
            }
        });
        newSelectPanel.add(optionBButton);

        JButton optionCButton = new JButton("普通");
        optionCButton.setAlignmentX(Component.CENTER_ALIGNMENT);
        optionCButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                showResult(optionType, "C");
            }
        });
        newSelectPanel.add(optionCButton);

        JButton optionDButton = new JButton("安い");
        optionDButton.setAlignmentX(Component.CENTER_ALIGNMENT);
        optionDButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                showResult(optionType, "D");
            }
        });
        newSelectPanel.add(optionDButton);

        // 新しいページのレイアウト設定
        newFrame.setLayout(new BorderLayout(5, 5));
        newFrame.add(newTitlePanel, BorderLayout.NORTH);
        newFrame.add(newFacePanel, BorderLayout.CENTER);
        newFrame.add(newSelectPanel, BorderLayout.SOUTH);

        newFrame.setVisible(true);

        // 新しいページで自動的に音声を再生
        playSound("9.wav");
    }

    private void playSound(String soundFileName) {
        try {
            AudioInputStream audioInputStream = AudioSystem.getAudioInputStream(new File(soundFileName));
            Clip clip = AudioSystem.getClip();
            clip.open(audioInputStream);
            clip.start();
        } catch (UnsupportedAudioFileException | IOException | LineUnavailableException e) {
            e.printStackTrace();
        }
    }
